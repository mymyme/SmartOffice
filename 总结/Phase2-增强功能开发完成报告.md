## 🔧 技术实现要点

### 1. 选中内容监听

```javascript
// 监听iframe内的选中事件
iframeDoc.addEventListener('mouseup', handlePreviewSelection);

function handlePreviewSelection(e) {
    const selection = iframeDoc.getSelection();
    const selectionText = selection.toString().trim();

    if (selectionText.length > 0) {
        // 保存Range对象
        selectedRange = selection.getRangeAt(0);
        // 显示工具栏
        showFloatingToolbar(e);
    }
}
```

### 2. 选中位置插入内容

```javascript
function insertImageAtSelection(imageHtml) {
    const tempDiv = iframeDoc.createElement('div');
    tempDiv.innerHTML = imageHtml;
    const imgElement = tempDiv.firstElementChild;

    // 删除选中内容并插入图片
    selectedRange.deleteContents();
    selectedRange.insertNode(imgElement);

    // 同步到编辑器
    syncPreviewToCode();
}
```

### 3. ZIP打包导出

```javascript
async function exportAsZIP() {
    const zip = new JSZip();

    // 下载所有图片
    for (const imgUrl of images) {
        const response = await fetch(imgUrl);
        const blob = await response.blob();
        zip.file(`images/${fileName}`, blob);
    }

    // 添加HTML
    zip.file('index.html', htmlWithLocalImages);

    // 生成并下载ZIP
    const content = await zip.generateAsync({
        type: 'blob',
        compression: 'DEFLATE'
    });

    downloadFile(content, 'export.zip');
}
```

### 4. Base64转换

```javascript
async function exportAsHTMLWithBase64() {
    for (const imgUrl of images) {
        const blob = await fetch(imgUrl).then(r => r.blob());

        // 检查大小限制
        if (blob.size > maxSize) continue;

        // 转换为Base64
        const base64 = await blobToBase64(blob);

        // 替换URL
        html = html.replace(imgUrl, base64);
    }
}
```
